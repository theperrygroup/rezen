    Use these imports for static type checking:

if TYPE_CHECKING:
    from rezen import RezenClient

    Runtime Validation¶

    Validate data at runtime:

    from rezen.enums import TeamStatus

def validate_status(status: str) -> bool:
    try:
        TeamStatus(status)
        return True
    except ValueError:
        return False
